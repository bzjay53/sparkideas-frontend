📖 사용법

const { RestAPIModule } = require('./shared-modules/api/express-rest'); const express = require('express'); const app = express(); app.use(express.json()); // 1. API 모듈 생성 const api = new RestAPIModule() .setDatabaseClient(dbClient) .createCRUDRoutes('ideas') .createCRUDRoutes('categories') .createAuthRoutes(); // 2. Express에 적용 app.use('/api', api.getRouter()); // 3. 서버 시작 app.listen(3000, () => { console.log('API Server running on port 3000'); });
